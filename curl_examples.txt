# Пример запроса для получения информации о текущем пользователе
GET /users/me
curl -X GET "http://localhost:8000/users/me?username=johndoe&password=testpassword" \
     -H "accept: application/json"

# Пример запроса для создания новой заметки
POST /notes/
curl -X POST "http://localhost:8000/notes/" \
     -H "accept: application/json" \
     -H "Content-Type: application/json" \
     -d '{
           "title": "New Note Title",
           "content": "This is the content of the new note."
         }'

# Пример запроса для получения списка заметок
GET /notes/
curl -X GET "http://localhost:8000/notes/" \
     -H "accept: application/json" \
     -H "Authorization: Bearer <your_token>"
